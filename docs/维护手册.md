##### 后端开发

```javascript
// === 步骤1: 创建数据文件 (backend/data/team.json) ===
[]

// === 步骤2: 创建路由 (backend/src/routes/team.js) ===
import express from 'express';
import { readData, writeData } from '../services/dataService.js';
import { authenticateToken } from '../middleware/auth.js';

const router = express.Router();

// GET 获取所有成员
router.get('/', async (req, res) => {
  const members = await readData('team');
  res.json(members || []);
});

// POST 添加成员（需要认证）
router.post('/', authenticateToken, async (req, res) => {
  const members = await readData('team') || [];
  const newMember = {
    id: Date.now().toString(),
    ...req.body,
    createdAt: new Date().toISOString()
  };
  members.push(newMember);
  await writeData('team', members);
  res.status(201).json(newMember);
});

// PUT 更新成员
router.put('/:id', authenticateToken, async (req, res) => {
  const members = await readData('team') || [];
  const index = members.findIndex(m => m.id === req.params.id);
  if (index !== -1) {
    members[index] = { ...members[index], ...req.body };
    await writeData('team', members);
    res.json(members[index]);
  } else {
    res.status(404).json({ error: '成员不存在' });
  }
});

// DELETE 删除成员
router.delete('/:id', authenticateToken, async (req, res) => {
  const members = await readData('team') || [];
  const filtered = members.filter(m => m.id !== req.params.id);
  await writeData('team', filtered);
  res.json({ message: '删除成功' });
});

export default router;

// === 步骤3: 注册路由 (backend/src/server.js) ===
// 添加导入：
import teamRoutes from './routes/team.js';

// 添加路由：
app.use('/api/team', teamRoutes);
```

## 附录：完整部署清单

### 首次部署步骤

```bash
# 1. 服务器环境准备
apt update && apt upgrade -y
apt install -y nodejs npm nginx git

# 2. 安装 PM2
npm install -g pm2

# 3. 克隆代码
cd /root
git clone https://github.com/xxx0922/personal-portfolio.git
cd personal-portfolio

# 4. 安装依赖
npm install
cd backend && npm install && cd ..

# 5. 构建前端
npm run build

# 6. 配置 Nginx
# 编辑 /etc/nginx/sites-available/default
# 配置反向代理和静态文件服务

# 7. 启动后端
cd backend
pm2 start src/server.js --name portfolio-backend
pm2 save
pm2 startup

# 8. 部署前端
cp -r dist/* /var/www/html/

# 9. 创建上传文件软链接
ln -s /root/personal-portfolio/backend/uploads /var/www/html/uploads

# 10. 重启服务
systemctl restart nginx
pm2 restart portfolio-backend

# 11. 验证
curl http://localhost:3001/api/health  # 后端健康检查
curl http://localhost/                 # 前端访问
```
